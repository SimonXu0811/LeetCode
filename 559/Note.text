自己的思路
递归的方式
遍历子树，如果不为null那么就+1，然后取所有子节点中最大的
最后再加上根节点的1

class Solution {
    public int maxDepth(Node root) {
        if (root == null) {
            return 0;
        }

        int max = 0;
        for (Node child : root.children) { //遍历所有子节点，有的话就加1
            int value = maxDepth(child);
            
            if (value > max) {//取最深的节点
                max = value;
            }
        }
        return max +1;//回溯
    }
}
